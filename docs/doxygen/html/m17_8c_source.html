22{
23 // Input validation
24 if (out == NULL || cnt == NULL) {
25 return;
26 }
27
28 if (*cnt >= SYM_PER_FRA) {
29 return; // Buffer overflow protection

32 if(type==PREAM_BERT) //pre-BERT
33 {

36 out[(*cnt)++]=-3.0;

38 out[(*cnt)++]=+3.0;

181void gen_frame(float out[SYM_PER_FRA], const uint8_t* data, const frame_t type, const lsf_t* lsf, const uint8_t lich_cnt, const uint16_t fn)
182{
183 uint8_t lich[6]; //48 bits packed raw, unencoded LICH
184 uint8_t lich_encoded[12]; //96 bits packed, encoded LICH
185 uint8_t enc_bits[SYM_PER_PLD*2]; //type-2 bits, unpacked
186 uint8_t rf_bits[SYM_PER_PLD*2]; //type-4 bits, unpacked
187 uint32_t sym_cnt=0; //symbols written counter
188
189 if(type==FRAME_LSF)
190 {
191 gen_syncword(out, &sym_cnt, SYNC_LSF);
192 conv_encode_LSF(enc_bits, lsf);
193 }
194 else if(type==FRAME_STR)
195 {
196 gen_syncword(out, &sym_cnt, SYNC_STR);
197 extract_LICH(lich, lich_cnt, lsf);
198 encode_LICH(lich_encoded, lich);
199 unpack_LICH(enc_bits, lich_encoded);
200 conv_encode_stream_frame(&enc_bits[96], data, fn); //stream frames require 16-byte payloads
201 }
202 else if(type==FRAME_PKT)
203 {
204 gen_syncword(out, &sym_cnt, SYNC_PKT);
205 conv_encode_packet_frame(enc_bits, data); //packet frames require 200-bit payload chunks plus a 6-bit counter
206 }
207 else if(type==FRAME_BERT)
208 {
209 gen_syncword(out, &sym_cnt, SYNC_BER);
210 conv_encode_bert_frame(enc_bits, data); //BERT frames require 197 BERT bits packed as 25 bytes
211 }
212
213 //common stuff
214 reorder_bits(rf_bits, enc_bits);
215 randomize_bits(rf_bits);
216 gen_data(out, &sym_cnt, rf_bits);
217}

274uint32_t decode_LSF(lsf_t* lsf, const float pld_symbs[SYM_PER_PLD])
275{
276 uint8_t lsf_b[30+1];
277 uint16_t soft_bit[2*SYM_PER_PLD];
278 uint16_t d_soft_bit[2*SYM_PER_PLD];
279 uint32_t e;
280
281 slice_symbols(soft_bit, pld_symbs);
282 randomize_soft_bits(soft_bit);
283 reorder_soft_bits(d_soft_bit, soft_bit);
284
285 e = viterbi_decode_punctured(lsf_b, d_soft_bit, puncture_pattern_1, 2*SYM_PER_PLD, sizeof(puncture_pattern_1));
286
287 //copy over the data starting at byte 1 (byte 0 needs to be omitted)
288 memcpy(lsf->dst, &lsf_b[1+0], 6); //DST field
289 memcpy(lsf->src, &lsf_b[1+6], 6); //SRC field
290 lsf->type[0]=lsf_b[1+12]; //TYPE field
291 lsf->type[1]=lsf_b[1+13];
292 memcpy(lsf->meta, &lsf_b[1+14], 14); //META field
293 lsf->crc[0]=lsf_b[1+28]; //CRC field
294 lsf->crc[1]=lsf_b[1+29];
295
296 return e; //return Viterbi error metric
297}

309uint32_t decode_str_frame(uint8_t frame_data[16], uint8_t lich[5], uint16_t* fn, uint8_t* lich_cnt, const float pld_symbs[SYM_PER_PLD])
310{
311 uint16_t soft_bit[2*SYM_PER_PLD];
312 uint16_t d_soft_bit[2*SYM_PER_PLD];
313 uint8_t tmp_frame_data[(16+128)/8+1]; //1 byte extra for flushing
314 uint32_t e;
315
316 slice_symbols(soft_bit, pld_symbs);
317 randomize_soft_bits(soft_bit);
318 reorder_soft_bits(d_soft_bit, soft_bit);
319
320 //decode LICH
321 uint8_t tmp[6];
322 decode_LICH(tmp, d_soft_bit);
323 memcpy(lich, tmp, 5);
324
325 if(lich_cnt!=NULL) *lich_cnt = tmp[5]>>5;
326
327 e = viterbi_decode_punctured(tmp_frame_data, &d_soft_bit[96], puncture_pattern_2, 2*SYM_PER_PLD-96, sizeof(puncture_pattern_2));
328
329 //shift 1+2 positions left - get rid of the encoded flushing bits and FN
330 memcpy(frame_data, &tmp_frame_data[1+2], 16);
331
332 if(fn!=NULL) *fn = (tmp_frame_data[1]<<8)|tmp_frame_data[2];
333
334 return e;
335}

346uint32_t decode_pkt_frame(uint8_t frame_data[25], uint8_t* eof, uint8_t* fn, const float pld_symbs[SYM_PER_PLD])
347{
348 uint16_t soft_bit[2*SYM_PER_PLD];
349 uint16_t d_soft_bit[2*SYM_PER_PLD];
350 uint8_t tmp_frame_data[26+1]; //1 byte extra for flushing
351 uint32_t e;
352
353 slice_symbols(soft_bit, pld_symbs);
354 randomize_soft_bits(soft_bit);
355 reorder_soft_bits(d_soft_bit, soft_bit);
356
357 e = viterbi_decode_punctured(tmp_frame_data, d_soft_bit, puncture_pattern_3, 2*SYM_PER_PLD, sizeof(puncture_pattern_3));

359 //shift 1 position left - get rid of the encoded flushing bits
