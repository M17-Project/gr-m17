74 if (secure_boot_init() != M17_SB_SUCCESS) {

99 status = M17_SB_ERROR_BOOT_CHAIN_INVALID;
100 }
101
102 // 4. Validate M17 library
103 if (status == M17_SB_SUCCESS &&
104 m17_sb_validate_component(M17_SB_COMPONENT_M17_LIBRARY) != M17_SB_SUCCESS) {
105 status = M17_SB_ERROR_BOOT_CHAIN_INVALID;
106 }
107
108 // 5. Validate crypto modules
109 if (status == M17_SB_SUCCESS &&
110 m17_sb_validate_component(M17_SB_COMPONENT_CRYPTO_MODULE) != M17_SB_SUCCESS) {
111 status = M17_SB_ERROR_BOOT_CHAIN_INVALID;
112 }

114 if (status == M17_SB_SUCCESS) {
115 g_sb_state.chain_valid = true;
116 // SECURITY FIX: Use secure timestamp generation
117 struct timespec ts;
118 if (clock_gettime(CLOCK_REALTIME, &ts) == 0) {

121 g_sb_state.last_validation = 0; // Fallback
122 }
123 g_sb_state.validation_count++;
124 }
