569 // scrambler pn sequence generation
570 void m17_coder_impl::scrambler_sequence_generator()
571 {
572 int i = 0;
573 uint32_t lfsr, bit;
574 lfsr = _scrambler_seed;
575
576 // only set if not initially set (first run), it is possible (and observed) that the scrambler_subtype can
577 // change on subsequent passes if the current SEED for the LFSR falls below one of these thresholds
578 if (_scrambler_subtype == -1)
579 {
580 if (lfsr > 0 && lfsr <= 0xFF)
581 _scrambler_subtype = 0; // 8-bit key
582 else if (lfsr > 0xFF && lfsr <= 0xFFFF)
583 _scrambler_subtype = 1; // 16-bit key
584 else if (lfsr > 0xFFFF && lfsr <= 0xFFFFFF)
585 _scrambler_subtype = 2; // 24-bit key
586 else
587 _scrambler_subtype = 0; // 8-bit key (default)
588 }
589
590 // TODO: Set Frame Type based on scrambler_subtype value
591 if (_debug == true)
592 {
593 // SECURITY FIX: Never log scrambler seeds or keys - they are sensitive cryptographic material
594 fprintf(stderr, "\nScrambler configured (details hidden for security); Subtype: %02d;\n", _scrambler_subtype);
595 fprintf(stderr, " pN: ");
596 }
597
598 // run pN sequence with taps specified
599 for (i = 0; i < 128; i++)
600 {
601 // get feedback bit with specified taps, depending on the scrambler_subtype
602 if (_scrambler_subtype == 0)
603 bit = (lfsr >> 7) ^ (lfsr >> 5) ^ (lfsr >> 4) ^ (lfsr >> 3);
604 else if (_scrambler_subtype == 1)
605 bit = (lfsr >> 15) ^ (lfsr >> 14) ^ (lfsr >> 12) ^ (lfsr >> 3);
606 else if (_scrambler_subtype == 2)
607 bit = (lfsr >> 23) ^ (lfsr >> 22) ^ (lfsr >> 21) ^ (lfsr >> 16);
608 else
609 bit = 0; // should never get here, but just in case
610
611 bit &= 1; // truncate bit to 1 bit (required since I didn't do it above)
612 lfsr = (lfsr << 1) | bit; // shift LFSR left once and OR bit onto LFSR's LSB
613 lfsr &= 0xFFFFFF; // truncate lfsr to 24-bit (really doesn't matter)
614 _scrambler_pn[i] = bit;
615 }
616 // pack bit array into byte array for easy data XOR
617 pack_bit_array_into_byte_array(_scrambler_pn, _scr_bytes, 16);
618
619 // save scrambler seed for next round
620 _scrambler_seed = lfsr;
621
622 // truncate seed so subtype will continue to set properly on subsequent passes
623 if (_scrambler_subtype == 0)
624 _scrambler_seed &= 0xFF;
625 else if (_scrambler_subtype == 1)
626 _scrambler_seed &= 0xFFFF;
627 else if (_scrambler_subtype == 2)
628 _scrambler_seed &= 0xFFFFFF;
629
630 if (_debug == true)
631 {
632 // SECURITY FIX: Removed scrambler bytes logging - sensitive material must never be printed
633 // for (i = 0; i < 16; i++)
634 // fprintf(stderr, " %02X", _scr_bytes[i]);
635 // fprintf(stderr, "\n");
636 }
637 }

639 // convert a user string (as hex octets) into a uint8_t array for key
640 void m17_coder_impl::parse_raw_key_string(uint8_t *dest,
641 const char *inp)
642 {
643 uint16_t len = strlen(inp);
644
645 if (len == 0)
646 return; // return silently and pretend nothing happened
647
648 memset(dest, 0, len / 2); // one character represents half of a byte
649
650 if (!(len % 2)) // length even?
651 {
