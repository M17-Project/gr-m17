55uint32_t viterbi_decode(uint8_t* out, const uint16_t* in, const uint16_t len)
56{
57 // Input validation
58 if (out == NULL || in == NULL) {
59 return 0xFFFFFFFF; // Error indicator
60 }
61
62 if (len == 0 || len > 244*2) {
63 return 0xFFFFFFFF; // Error indicator
64 }
65
66 // Initialize state if needed
67 m17_error_t err = viterbi_init_state();
68 if (err != M17_SUCCESS) {
69 return 0xFFFFFFFF; // Error indicator
70 }
71
72 M17_MUTEX_LOCK(viterbi_mutex);
73
74 viterbi_reset();
75
76 size_t pos = 0;
77 for(size_t i = 0; i < len; i += 2)
78 {
79 uint16_t s0 = in[i];
80 uint16_t s1 = in[i + 1];
81
82 viterbi_decode_bit(s0, s1, pos);
83 pos++;
84 }
85
86 uint32_t result = viterbi_chainback(out, pos, len/2);
87
88 M17_MUTEX_UNLOCK(viterbi_mutex);
89
90 return result;
91}

103uint32_t viterbi_decode_punctured(uint8_t* out, const uint16_t* in, const uint8_t* punct, const uint16_t in_len, const uint16_t p_len)
104{
105 // Input validation
106 if (out == NULL || in == NULL || punct == NULL) {
107 return 0xFFFFFFFF; // Error indicator
108 }
109
110 if (in_len == 0 || in_len > 244*2 || p_len == 0) {
111 return 0xFFFFFFFF; // Error indicator
112 }
113
114 uint16_t umsg[244*2]; //unpunctured message
115 uint8_t p=0; //puncturer matrix entry
116 uint16_t u=0; //bits count - unpunctured message
117 uint16_t i=0; //bits read from the input message
118
119 while(i<in_len)
120 {
121 if(punct[p])
122 {
123 umsg[u]=in[i];
124 i++;
125 }
126 else
127 {
128 umsg[u]=0x7FFF;
129 }
130
131 u++;
132 p++;
133 p%=p_len;
134 }
135
136 return viterbi_decode(out, umsg, u) - (u-in_len)*0x7FFF;
137}

146void viterbi_decode_bit(uint16_t s0, uint16_t s1, const size_t pos)
147{
148 if (viterbi_state == NULL) {
149 return; // State not initialized
150 }
151
152 static const uint16_t COST_TABLE_0[] = {0, 0, 0, 0, 0xFFFF, 0xFFFF, 0xFFFF, 0xFFFF};
153 static const uint16_t COST_TABLE_1[] = {0, 0xFFFF, 0xFFFF, 0, 0, 0xFFFF, 0xFFFF, 0};
154
155 for(uint8_t i = 0; i < M17_CONVOL_STATES/2; i++)
156 {
157 uint32_t metric = q_abs_diff(COST_TABLE_0[i], s0)
158 + q_abs_diff(COST_TABLE_1[i], s1);
159
160
161 uint32_t m0 = viterbi_state->prevMetrics[i] + metric;
162 uint32_t m1 = viterbi_state->prevMetrics[i + M17_CONVOL_STATES/2] + (0x1FFFE - metric);
163
164 uint32_t m2 = viterbi_state->prevMetrics[i] + (0x1FFFE - metric);
165 uint32_t m3 = viterbi_state->prevMetrics[i + M17_CONVOL_STATES/2] + metric;
166
167 uint8_t i0 = 2 * i;
168 uint8_t i1 = i0 + 1;
169
170 if(m0 >= m1)
171 {
172 viterbi_state->viterbi_history[pos]|=(1<<i0);
173 viterbi_state->currMetrics[i0] = m1;
174 }
175 else
176 {
177 viterbi_state->viterbi_history[pos]&=~(1<<i0);
178 viterbi_state->currMetrics[i0] = m0;
179 }
180
181 if(m2 >= m3)
182 {
183 viterbi_state->viterbi_history[pos]|=(1<<i1);
184 viterbi_state->currMetrics[i1] = m3;
185 }
186 else
187 {
188 viterbi_state->viterbi_history[pos]&=~(1<<i1);
189 viterbi_state->currMetrics[i1] = m2;
190 }
191 }
192
193 //swap
194 uint32_t tmp[M17_CONVOL_STATES];
195 for(uint8_t i=0; i<M17_CONVOL_STATES; i++)
196 {
197 tmp[i]=viterbi_state->currMetrics[i];
198 }
199 for(uint8_t i=0; i<M17_CONVOL_STATES; i++)
200 {
201 viterbi_state->currMetrics[i]=viterbi_state->prevMetrics[i];
202 viterbi_state->prevMetrics[i]=tmp[i];
203 }
204}

214uint32_t viterbi_chainback(uint8_t* out, size_t pos, uint16_t len)
215{
216 if (viterbi_state == NULL) {
217 return 0xFFFFFFFF; // Error indicator
218 }
219
220 uint8_t state = 0;
221 size_t bitPos = len+4;
222
223 memset(out, 0, (bitPos/8)+1);
224
225 while(pos > 0)
226 {
227 bitPos--;
228 pos--;
229 uint16_t bit = viterbi_state->viterbi_history[pos]&((1<<(state>>4)));
230 state >>= 1;
231 if(bit)
232 {
233 state |= 0x80;
234 out[bitPos/8]|=1<<(7-(bitPos%8));
235 }
236 }
237
238 uint32_t cost = viterbi_state->prevMetrics[0];
239
240 for(size_t i = 0; i < M17_CONVOL_STATES; i++)
241 {
242 uint32_t m = viterbi_state->prevMetrics[i];
243 if(m < cost) cost = m;
244 }
245
246 return cost;
247}

253void viterbi_reset(void)
254{
255 if (viterbi_state == NULL) {
256 return; // State not initialized
257 }
258
259 memset((uint8_t*)viterbi_state->viterbi_history, 0, 2*244);
260 memset((uint8_t*)viterbi_state->currMetrics, 0, 4*M17_CONVOL_STATES);
261 memset((uint8_t*)viterbi_state->prevMetrics, 0, 4*M17_CONVOL_STATES);
262 memset((uint8_t*)viterbi_state->currMetricsData, 0, 4*M17_CONVOL_STATES);
263 memset((uint8_t*)viterbi_state->prevMetricsData, 0, 4*M17_CONVOL_STATES);
264}
