25// Generate unique session ID
26static int generate_session_id(uint8_t session_id[16]) {
27 FILE *urandom = fopen("/dev/urandom", "rb");
28 if (urandom == NULL) {
29 return -1;
30 }
31
32 size_t bytes_read = fread(session_id, 1, 16, urandom);
33 fclose(urandom);
34
35 if (bytes_read != 16) {
36 return -1;
37 }
38
39 return 0;
40}

79// Derive session key from ECDH shared secret
80int m17_derive_session_key(const uint8_t *shared_secret, size_t shared_secret_len,
81 const m17_key_context_t *context,
82 uint8_t *derived_key, size_t key_len) {
83 if (shared_secret == NULL || context == NULL || derived_key == NULL) {
84 return -1;
85 }
86
87 if (shared_secret_len == 0 || key_len == 0) {
88 return -1;
89 }
90
91 // Create context string for HKDF
92 uint8_t context_data[64];
93 size_t context_len = 0;
94
95 // Add session ID
96 memcpy(context_data + context_len, context->session_id, 16);
97 context_len += 16;
98
99 // Add timestamp
100 memcpy(context_data + context_len, &context->timestamp, 8);
101 context_len += 8;
102
103 // Add sender callsign
104 size_t sender_len = strlen((char*)context->sender_callsign);
105 memcpy(context_data + context_len, context->sender_callsign, sender_len);
106 context_len += sender_len;
107
108 // Add receiver callsign
109 size_t receiver_len = strlen((char*)context->receiver_callsign);
110 memcpy(context_data + context_len, context->receiver_callsign, receiver_len);
111 context_len += receiver_len;
112
113 // Add frame number
114 memcpy(context_data + context_len, &context->frame_number, 2);
115 context_len += 2;
116
117 // Add key type
118 context_data[context_len] = context->key_type;
119 context_len++;
120
121 // Use HKDF to derive key
122 int result = m17_hkdf_derive(shared_secret, shared_secret_len,
123 context_data, context_len,
124 (uint8_t*)"M17-Session-Key", 15,
125 derived_key, key_len);
126
127 return result;
128}

130// Derive multiple keys for a session
131int m17_derive_session_keys(const uint8_t *shared_secret, size_t shared_secret_len,
132 const m17_key_context_t *context,
133 uint8_t *encryption_key, size_t enc_key_len,
134 uint8_t *authentication_key, size_t auth_key_len,
135 uint8_t *integrity_key, size_t int_key_len) {
136 if (shared_secret == NULL || context == NULL) {
137 return -1;
138 }
139
140 int result = 0;
141
142 // Derive encryption key
143 if (encryption_key != NULL && enc_key_len > 0) {
144 m17_key_context_t enc_ctx = *context;
145 enc_ctx.key_type = M17_KEY_TYPE_ENCRYPTION;
146 result |= m17_derive_session_key(shared_secret, shared_secret_len,
147 &enc_ctx, encryption_key, enc_key_len);
148 }
149
150 // Derive authentication key
151 if (authentication_key != NULL && auth_key_len > 0) {
152 m17_key_context_t auth_ctx = *context;
153 auth_ctx.key_type = M17_KEY_TYPE_AUTHENTICATION;
154 result |= m17_derive_session_key(shared_secret, shared_secret_len,
155 &auth_ctx, authentication_key, auth_key_len);
156 }
157
158 // Derive integrity key
159 if (integrity_key != NULL && int_key_len > 0) {
160 m17_key_context_t int_ctx = *context;
161 int_ctx.key_type = M17_KEY_TYPE_INTEGRITY;
162 result |= m17_derive_session_key(shared_secret, shared_secret_len,
163 &int_ctx, integrity_key, int_key_len);
164 }
165
166 return result;
167}

169// Verify key derivation context
170bool m17_verify_key_context(const m17_key_context_t *context) {
171 if (context == NULL) {
172 return false;
173 }
174
175 // Check timestamp is reasonable (not too old, not in future)
176 // SECURITY FIX: Use secure timestamp generation
177 struct timespec ts;
178 uint64_t current_time;
179 if (clock_gettime(CLOCK_REALTIME, &ts) == 0) {
180 current_time = (uint64_t)ts.tv_sec;
181 } else {
182 current_time = 0; // Fallback
183 }
184 if (context->timestamp > current_time ||
185 (current_time - context->timestamp) > 3600) { // 1 hour max
186 return false;
187 }
188
189 // Check frame number is valid
190 if (context->frame_number > 0xFFFF) {
191 return false;
192 }
193
194 // Check key type is valid
195 if (context->key_type < 1 || context->key_type > 4) {
196 return false;
197 }
198
199 // Check callsigns are not empty
200 if (strlen((char*)context->sender_callsign) == 0 ||
201 strlen((char*)context->receiver_callsign) == 0) {
202 return false;
203 }
204
205 return true;
206}

208// Secure key comparison (constant-time)
209bool m17_secure_key_compare(const uint8_t *key1, const uint8_t *key2, size_t len) {
210 if (key1 == NULL || key2 == NULL || len == 0) {
211 return false;
212 }
213
214 uint8_t result = 0;
215 for (size_t i = 0; i < len; i++) {
216 result |= (key1[i] ^ key2[i]);
217 }
218
219 return (result == 0);
220}

222// Secure key wiping
223void m17_secure_key_wipe(uint8_t *key, size_t len) {
224 if (key == NULL || len == 0) {
225 return;
226 }
227
228 // Multiple passes with different patterns
229 volatile uint8_t *volatile_key = (volatile uint8_t *)key;
230
231 for (size_t i = 0; i < len; i++) volatile_key[i] = 0x00;
232 for (size_t i = 0; i < len; i++) volatile_key[i] = 0xFF;
233 for (size_t i = 0; i < len; i++) volatile_key[i] = 0xAA;
234 for (size_t i = 0; i < len; i++) volatile_key[i] = 0x55;
235 for (size_t i = 0; i < len; i++) volatile_key[i] = 0x00;
236
237 // Memory barrier
238 __asm__ __volatile__("" ::: "memory");
239}
